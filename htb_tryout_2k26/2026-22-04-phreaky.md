now run one python script in that directory which is given below.  
```python
import os
import re
import email
import subprocess

def solve_challenge():
    # 1. Find all exported email files in the current directory
    eml_files = [f for f in os.listdir('.') if f.endswith('.eml')]
    
    if not eml_files:
        print("[-] No .eml files found. Run tshark --export-objects imf,ext first.")
        return

    print(f"[+] Found {len(eml_files)} email files. Commencing extraction...")

    # 2. Iterate through each email
    for eml_file in eml_files:
        with open(eml_file, 'r') as f:
            msg = email.message_from_file(f)
        
        password = None
        zip_payload = None
        
        # 3. Walk the MIME parts to isolate the text (password) and application/zip (payload)
        for part in msg.walk():
            content_type = part.get_content_type()
            
            if content_type == 'text/plain':
                body = part.get_payload(decode=True).decode('utf-8', errors='ignore')
                # Extract the password using a regex group
                match = re.search(r'Password:\s*(\S+)', body)
                if match:
                    password = match.group(1)
            
            elif content_type == 'application/zip':
                # decode=True automatically handles the Base64 decoding to raw bytes
                zip_payload = part.get_payload(decode=True)
                
        # 4. If we successfully grabbed both the lock and the key, crack it open
        if password and zip_payload:
            temp_zip = "temp_chunk.zip"
            with open(temp_zip, 'wb') as zf:
                zf.write(zip_payload)
            
            print(f"[>] Decrypting payload from {eml_file} with key: {password}")
            # We use subprocess to call the system 'unzip' as it handles legacy PKZIP crypto better than Python's zipfile module
            subprocess.run(['unzip', '-P', password, '-o', temp_zip], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
            
            # Clean up the temporary zip file
            os.remove(temp_zip)

    # 5. Reassemble the final PDF by concatenating in explicit numerical order
    print("\n[+] All fragments extracted. Reconstructing final PDF...")
    
    with open("final_flag.pdf", "wb") as final_pdf:
        for i in range(1, 16):
            part_filename = f"phreaks_plan.pdf.part{i}"
            if os.path.exists(part_filename):
                with open(part_filename, "rb") as fragment:
                    final_pdf.write(fragment.read())
            else:
                print(f"[-] WARNING: Missing {part_filename}! The PDF might be corrupted.")

    print("[+] Success! The payload has been fully reassembled into 'final_flag.pdf'")

if __name__ == '__main__':
    solve_challenge()
```
